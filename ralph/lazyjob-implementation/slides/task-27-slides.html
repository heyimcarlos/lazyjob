Enter Key Fix

From broken catch-all to proper job opening

Before (Broken)

match code {
    KeyCode::Char('/') => { /* search */ }
    KeyCode::Char('f') => { /* filter */ }
    KeyCode::Char('s') => { /* sort */ }
    KeyCode::Down => { /* scroll */ }
    KeyCode::Up => { /* scroll */ }
    KeyCode::Enter | _ => None,  // ← catch-all swallows Enter!
}

After (Fixed)

match code {
    // ... same as above ...
    KeyCode::Enter => {
        self.selected_job()
            .map(|job| Action::OpenJob(job.id))
    }
    _ => None,
}

Database Loading

Wiring JobRepository to the TUI

App::load_jobs()

pub async fn load_jobs(&mut self) {
    let Some(pool) = &self.pool else { return };
    let repo = JobRepository::new(pool.clone());
    match repo.list(&Pagination::default()).await {
        Ok(jobs) => self.views.jobs_list.set_jobs(jobs),
        Err(err) => tracing::warn!("Failed to load jobs: {err}"),
    }
}

Architecture

Data flow for the Jobs List

Key Press Flow

User presses 'j'
  -> event_loop: keymap.resolve(Jobs, j) = Action::ScrollDown
  -> app.handle_action(ScrollDown)
  -> active_view_mut().handle_key(Down, NONE)
  -> jobs_list.scroll_down()

User presses '/'
  -> event_loop: keymap returns None (/ not in keymap)
  -> falls through to: active_view_mut().handle_key('/', NONE)
  -> jobs_list returns Action::EnterSearch
  -> app sets input_mode = Search
  -> subsequent keys bypass keymap, go directly to view

User presses Enter (via Select action)
  -> app.handle_action(Select)
  -> active_view_mut().handle_key(Enter, NONE)
  -> jobs_list returns Action::OpenJob(selected_job.id)
  -> app.handle_action(OpenJob(_)) -> no-op (task 28)

Data Loading Flow

TUI startup
  -> lib.rs: Database::connect()
  -> App::new().with_pool(pool)
  -> app.load_jobs().await
  -> JobRepository::list(Pagination::default())
  -> view.set_jobs(jobs)
  -> apply_filter_sort() -> filtered indices ready
  -> render: Table with 6 columns
